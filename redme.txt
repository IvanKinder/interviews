python3 manage.py runserver  # старт локального сервера

http://127.0.0.1:8000/manage/poll/create/  # создание опроса (доступно только админу)
http://127.0.0.1:8000/manage/poll/update/<pk>/  # редактирование опроса (доступно только админу)
http://127.0.0.1:8000/manage/poll/list/  # список опросов (доступно только авторизованному юзеру)
http://127.0.0.1:8000/manage/question/create/  # создание вопроса (доступно только админу)
http://127.0.0.1:8000/manage/question/update/<pk>/  # редактирование вопроса (доступно только админу)
http://127.0.0.1:8000/manage/poll/<pk>/  # список вопросов в опросе (доступно только авторизованному юзеру)
http://127.0.0.1:8000/manage/answer/  # внести ответ на вопрос (доступно только авторизованному юзеру)
http://127.0.0.1:8000/manage/answer/list/  # список ответов на вопросы (доступно только авторизованному юзеру)
http://127.0.0.1:8000/manage/answer/poll/  # список опросов, в которых участовал юзер  (доступно только авторизованному юзеру)
